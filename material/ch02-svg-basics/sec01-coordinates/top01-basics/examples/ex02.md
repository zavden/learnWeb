# HTML

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Rectangle Coordinates</title>
</head>
<body>
    <svg width="400" height="200" class="svg-container">
        <!-- Grid for reference -->
        <defs>
            <pattern id="grid" width="10" height="10" patternUnits="userSpaceOnUse">
                <path d="M 20 0 L 0 0 0 20" fill="none" stroke="gray" stroke-width="0.5"/>
            </pattern>
        </defs>
        <rect width="100%" height="100%" fill="url(#grid)" />

        <!-- Rectangle at x=50, y=50 -->
        <rect x="50" y="50" width="100" height="80" fill="coral" />
        
        <!-- Text label -->
        <text x="50" y="40" font-family="monospace" font-size="12">x=50, y=50</text>
    </svg>
</body>
</html>
```
