<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Polygon Points</title>

<body>
    <svg width="300" height="300" class="svg-container">
        <!-- Triangle defined by 3 points -->
        <polygon points="150,20 250,250 50,250" fill="gold" stroke="orange" stroke-width="3" />
        
        <!-- Coordinates for vertices -->
        <text x="150" y="15" text-anchor="middle">150,20</text>
        <text x="260" y="250">250,250</text>
        <text x="40" y="250" text-anchor="end">50,250</text>
    </svg>
</body>
</html>
