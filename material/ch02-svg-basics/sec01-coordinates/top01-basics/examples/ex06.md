# HTML

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Coordinate Transformation (Translate)</title>
</head>
<body>
    <svg width="400" height="300" class="svg-container">
        <!-- Original coordinate system (0,0) -->
        <rect x="0" y="0" width="50" height="50" fill="#ccc" opacity="0.5" />
        <text x="5" y="20" font-size="10">Origin</text>

        <!-- Transformed Group -->
        <!-- Moves the coordinate system for children by 100, 100 -->
        <g transform="translate(100, 100)">
            <!-- This rect is at 0,0 relative to the group, but 100,100 on canvas -->
            <rect x="0" y="0" width="100" height="100" fill="teal" />
            <text x="5" y="20" fill="white">0,0 (local)</text>
            <circle cx="0" cy="0" r="5" fill="red" />
        </g>
    </svg>
</body>
</html>
```

# CSS

```css

    border: 1px solid black;
}
```

# JavaScript

```javascript
console.log('The <g> element with translate(x,y) shifts the origin (0,0) for all its children.');
```
