# HTML

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Nested SVG Coordinates</title>
</head>
<body>
    <!-- Main SVG -->
    <svg width="400" height="400" class="svg-container">
        <rect width="100%" height="100%" fill="#eee" />
        <text x="10" y="20">Root SVG (0,0)</text>

        <!-- Nested SVG at 50,50 with its own viewBox -->
        <svg x="50" y="50" width="100" height="100" viewBox="0 0 10 10">
            <!-- In this nested coordinate system, 10 units = 100px -->
            <!-- So 1 unit = 10px -->
            <rect width="10" height="10" fill="purple" />
            <circle cx="5" cy="5" r="2" fill="yellow" />
        </svg>
    </svg>
</body>
</html>
```
