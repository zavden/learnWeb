# HTML

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Interactive Coordinates</title>
</head>
<body>
    <svg id="mysvg" width="400" height="300" viewBox="0 0 400 300" class="svg-container">
        <text x="10" y="30" font-family="monospace">Move mouse over SVG</text>
        
        <!-- Crosshairs -->
        <line id="x-line" x1="0" y1="0" x2="400" y2="0" stroke="red" stroke-dasharray="4" style="display:none" />
        <line id="y-line" x1="0" y1="0" x2="0" y2="300" stroke="red" stroke-dasharray="4" style="display:none" />
        
        <!-- Coordinate Text -->
        <text id="coords" x="0" y="0" font-size="12" fill="red" style="display:none"></text>
    </svg>
</body>
</html>
```

# JavaScript

```javascript
const svg = document.getElementById('mysvg');
const pt = svg.createSVGPoint(); // Create a point for coordinate transformation
const xLine = document.getElementById('x-line');
const yLine = document.getElementById('y-line');
const coordsText = document.getElementById('coords');

svg.addEventListener('mousemove', (evt) => {
    pt.x = evt.clientX;
    pt.y = evt.clientY;

    // Convert screen coordinates to SVG coordinates
    const cursorPt = pt.matrixTransform(svg.getScreenCTM().inverse());
    
    const x = Math.round(cursorPt.x);
    const y = Math.round(cursorPt.y);

    // Update UI
    xLine.style.display = 'block';
    yLine.style.display = 'block';
    coordsText.style.display = 'block';
    
    xLine.setAttribute('y1', y);
    xLine.setAttribute('y2', y);
    
    yLine.setAttribute('x1', x);
    yLine.setAttribute('x2', x);
    
    coordsText.textContent = `x:${x}, y:${y}`;
    coordsText.setAttribute('x', x + 10);
    coordsText.setAttribute('y', y - 10);
});
```
