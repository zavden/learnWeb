# HTML

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Z-Index Stacking</title>
</head>
<body>
    <div class="container">
        <div class="card card-1">1</div>
        <div class="card card-2">2</div>
        <div class="card card-3">3</div>
    </div>
</body>
</html>
```

# CSS

```css
.container {
    position: relative;
    width: 300px;
    height: 200px;
    margin: 50px;
}

.card {
    width: 100px;
    height: 100px;
    position: absolute;
    border: 2px solid white;
    color: white;
    font-size: 24px;
    display: flex;
    align-items: center;
    justify-content: center;
}

.card-1 {
    background-color: red;
    top: 0;
    left: 0;
    z-index: 10;
}

.card-2 {
    background-color: blue;
    top: 30px;
    left: 30px;
    z-index: 20; /* Higher z-index appears on top */
}

.card-3 {
    background-color: green;
    top: 60px;
    left: 60px;
    z-index: 5; /* Lower z-index appears below others */
}
```

# JavaScript

```javascript
// Toggle z-index on click
const cards = document.querySelectorAll('.card');
cards.forEach(card => {
    card.addEventListener('click', () => {
        // Bring clicked card to front
        card.style.zIndex = 100;
    });
});
```
